YAML ​

yml
# 🌰 基本例子
---
#  : 代表对象
languages:
  # - 代表数组
  - Ruby
    Perl
  - Python
websites:
  date: 1976-07-31
  str: 这是一行字符串
  bool: true
  # 注释：~ 代表null
  parent: ~
  # 注释：!! 代表强制转换数据类型
  f: !!str true

# & 用来建立锚点（defaults）* 用来引用锚点 << 表示合并到当前
defaults: &defaults
  adapter: postgres
  host: localhost

development:
  database: myapp_development
  <<: *defaults
